PREFIX ns1: <http://example.org/>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT DISTINCT ?course ?event ?frequency
WHERE {
  ?course ns1:hasCourseCode "COMP" ;
          ns1:hasCourseNumber "474" .
  ?topic rdfs:label "rdf" .
  ?event ns1:contentTopic ?topic ;
         ns1:contentFor ?course .
  {
    SELECT ?event (COUNT(?event) as ?frequency)
    WHERE {
      ?event ns1:contentTopic ?topic ;
             ns1:contentFor ?course .
    }
    GROUP BY ?event
  }
}
ORDER BY DESC(?frequency)
